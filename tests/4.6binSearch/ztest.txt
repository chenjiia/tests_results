//8 bit ver
var S = 0;
var O = 0;

var SMax = 256;
var OMax = 256;

var STemp = 0;
var OTemp = 0;
var OCounter = 0;
var temp1 = 0;

while(O != OMax){
	S = 0;
	while(S != SMax){
		//code start
		OTemp = 0;
		STemp = S;
		
		temp1 = OTemp + 128;
		if(temp1 <= STemp){
			OTemp = OTemp + 128;
		}		
		temp1 = OTemp + 64;
		if(temp1 <= STemp){
			OTemp = OTemp + 64;
		}
		temp1 = OTemp + 32;
		if(temp1 <= STemp){
			OTemp = OTemp + 32;
		}
		temp1 = OTemp + 16;
		if(temp1 <= STemp){
			OTemp = OTemp + 16;
		}
		

		//code end
		if(O == OTemp){
			OCounter = OCounter + 1;
			goto break;
		}
		S = S + 1;
	}
	break : O = O + 1;
}
if(OCounter == 16){
	reach: 0>1;
}
