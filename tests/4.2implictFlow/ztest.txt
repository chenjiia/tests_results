var S = 0;
var O = 0;

var SMax = 256;
var OMax = 256;

var STemp = 0;
var OTemp = 0;
var OCounter = 0;


while(O != OMax){
	S = 0;
	while(S != SMax){
		//code start
		if(S == 0){
			OTemp = 0;
		}else{
			if(S == 1){
				OTemp = 1;
			}else{
				if(S == 2){
					OTemp = 2;
				}else{
					if(S == 3){
						OTemp = 3;
					}else{
						if(S == 4){
							OTemp = 4;
						}else{
							if(S == 5){
								OTemp = 5;
							}else{
								if(S == 6){
									OTemp = 6;
								}
							}
						}
					}
				}
			}
		}
		//code end
		if(O == OTemp){
			OCounter = OCounter + 1;
			goto break;
		}
		S = S + 1;
	}
	break : O = O + 1;
}

if(OCounter == 7){
	reach: 0>1;
}
